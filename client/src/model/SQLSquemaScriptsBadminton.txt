create table partida (
 	pk_partida int unique not null,
 	totalPontos int not null,
 	constraint pk_partida primary key (pk_partida)	
);

create table area1 (
	fk_partida int unique not null,
	saque int not null,
    forehand int not null,
    backhand int not null,
    clear int not null,
    drop_ int not null,
    smash int not null,
    drive int not null,
    lob int not null,
    netShot int not null,
    netKill int not null,
    netLift int not null,
    ponto int not null,
	FOREIGN KEY (fk_partida) REFERENCES partida(pk_partida)
);
